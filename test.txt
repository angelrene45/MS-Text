def get_host_details():
    hostname = socket.gethostname()
    total_memory = psutil.virtual_memory().total / (1024**3)  # Total memory in GB
    total_cpu = psutil.cpu_count(logical=True)
    cpu_frequency = psutil.cpu_freq().current if psutil.cpu_freq() else "N/A"
    
    return {
        "hostname": hostname,
        "total_memory_gb": total_memory,
        "total_cpu": total_cpu,
        "cpu_frequency_mhz": cpu_frequency,
        "uptime_seconds": time.time() - psutil.boot_time()
    }
